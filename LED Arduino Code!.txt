LEDs brightness code!

int ledPin = 9; // Pin connected to the LED
int brightness = 0; // Initial brightness
int fadeAmount = 5; // Amount to change the brightness by

void setup() {
  pinMode(ledPin, OUTPUT);
}

void loop() {
  analogWrite(ledPin, brightness); // Set the brightness of the LED

  brightness = brightness + fadeAmount; // Change the brightness for next time

  if (brightness <= 0 || brightness >= 255) {
    fadeAmount = -fadeAmount; // Reverse the direction of the fade
  }

  delay(30); // Wait for 30 milliseconds to see the dimming effect
}

Basic Arduino Code to Run a Stepper Motor!


#include <Stepper.h>

const int stepsPerRevolution = 200; // Number of steps per revolution (depends on your motor)

Stepper myStepper(stepsPerRevolution, 8, 9, 10, 11); // Pins connected to the stepper motor

void setup() {
  myStepper.setSpeed(60); // Set the speed to 60 RPM
}

void loop() {
  myStepper.step(stepsPerRevolution); // Move the motor one full revolution
  delay(1000); // Wait for 1 second
  myStepper.step(-stepsPerRevolution); // Move the motor in the opposite direction
  delay(1000); // Wait for 1 second
}
